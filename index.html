C# Constructors

    public string model;

        model = modelName;

C# Inheritance

class Animal {
    public void MakeSound() {
        Console.WriteLine("Animal Sound");
    }
}
class Dog : Animal {}
Dog myDog = new Dog();
myDog.MakeSound();

C# Interface

interface IAnimal {
    void MakeSound();
}
class Dog : IAnimal {
    public void MakeSound() {
        Console.WriteLine("Bark");
    }
}
Dog myDog = new Dog();
myDog.MakeSound();
